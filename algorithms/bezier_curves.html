Drawing bezier curves in a linear framebuffer

Cyao

2023

Drawing bezier curves onto a linear framebuffer is often a hard task for programmers. Here, I will try to make the simplest approche to draw them onto the screen.

On computers, we draw bezier curves as a polygon, which is, a series of lines. To draw it, we need to check if the line is flat enough, if yes, draw it using a line, if not, biscut the line in half.

A quadratic bezier curve is a bezier curve with 3 control points, cubic bezier is the same as the quadratic one, but with 4 control points. We can declare them in a struct like this: 

typedef struct Vector2 { 
    uint16_t x, y; 
} Vector2; 
 
typedef struct Qbezier { 
    Vector2 p0, p1, p2; 
} Qbezier; 
 
typedef struct Cbezier { 
    Vector2 p0, p1, p2, p3; 
} Qbezier;

Now we need to make the function to draw a line, like this: 

// A algorithm to draw lines, forgot what it's called ¯\_("/)_/¯ 
void line(Vector2 p0, Vector2 p1, const uint16_t color) { 
    bool yLonger = false; 
    int32_t shortLen = (int32_t) (p1.y - p0.y); 
    int32_t longLen = (int32_t) (p1.x - p0.x); 
     
    if (shortLen > longLen) { 
        int swap = shortLen; 
        shortLen = longLen; 
        longLen = swap; 
        yLonger = true; 
    } 
    int decInc; 
    if (longLen == 0) 
        decInc = 0; 
    else 
        decInc = (shortLen << 16) / longLen; 
 
    if (yLonger) { 
        if (longLen > 0) { 
            longLen += (int32_t) p0.y; 
            for (int j = (int32_t) (0x8000 + (p0.x << 16)); (int32_t) p0.y <= longLen; ++p0.y) { 
                put_pixel(j >> 16, p0.y, color); 
                j += decInc; 
            } 
            return; 
        } 
        longLen += (int32_t) p0.y; 
        for (int j = (int32_t) (0x8000 + (p0.x << 16)); (int32_t) p0.y >= longLen; --p0.y) { 
            put_pixel(j >> 16, p0.y, color); 
            j -= decInc; 
        } 
        return; 
    } 
 
    if (longLen > 0) { 
        longLen += (int32_t) p0.x; 
        for (int j = (int32_t) (0x8000 + (p0.y << 16)); (int32_t) p0.x <= longLen; ++p0.x) { 
            put_pixel(p0.x, j >> 16, color); 
            j += decInc; 
        } 
        return; 
    } 
    longLen += (int32_t) p0.x; 
    for (int j = (int32_t) (0x8000 + (p0.y << 16)); (int32_t) p0.x >= longLen; --p0.x) { 
        put_pixel(p0.x, j >> 16, color); 
        j -= decInc; 
    } 
}

Now that we have a function to draw a line, we need to implement our algorithm to biscut line in half. 

static Vector2 midpoint(Vector2 p1, Vector2 p2) { 
    return (Vector2) {.x = (int) (p2.x - p1.x) / 2 + p1.x, .y = (int) (p2.y - p1.y) / 2 + p1.y}; 
} 
 
// Takes in 1 bezier curve and splits them in half 
static void QSplitCurve(const Vector2 p[3], Vector2 out[2][3]) { 
    const Vector2 p12 = midpoint(p[0], p[1]); 
    const Vector2 p23 = midpoint(p[1], p[2]); 
    const Vector2 p123 = midpoint(p12, p23); 
    out[0][0] = p[0]; 
    out[0][1] = p12; 
    out[0][2] = p123; 
    out[1][0] = p123; 
    out[1][1] = p23; 
    out[1][2] = p[2]; 
} 
 
static void CSplitCurve(const Vector2 p[4], Vector2 out[2][4]) { 
    const Vector2 p12 = midpoint(p[0], p[1]); 
    const Vector2 p23 = midpoint(p[1], p[2]); 
    const Vector2 p34 = midpoint(p[3], p[2]); 
    const Vector2 p123 = midpoint(p12, p23); 
    const Vector2 p234 = midpoint(p23, p34); 
    const Vector2 p1234 = midpoint(p123, p234); 
    out[0][0] = p[0]; 
    out[0][1] = p12; 
    out[0][2] = p123; 
    out[0][3] = p1234; 
    out[1][0] = p1234; 
    out[1][1] = p234; 
    out[1][2] = p34; 
    out[1][3] = p[3]; 
}

Now that we know how to split the curves, we need to evaluate if the curve is flat enough. For the qbezier curve we just need the distence between the b(0.5) and the midpoint of P0 and P3. For the cubic bezier curve, it is more compicated, I am using the algorith from the pdf Piecewise Linear Approximation of Bezier Curves from Kaspar Fischer. 

static uint32_t CFlatness(const Vector2 p[4]) { 
    const Vector2 p1 = p[0]; 
    const Vector2 cp1 = p[1]; 
    const Vector2 cp2 = p[2]; 
    const Vector2 p2 = p[3]; 
    uint32_t ux = (3 * cp1.x - 2 * p1.x - p2.x) * (3 * cp1.x - 2 * p1.x - p2.x); 
    uint32_t uy = (3 * cp1.y - 2 * p1.y - p2.y) * (3 * cp1.y - 2 * p1.y - p2.y); 
    uint32_t vx = (3 * cp2.x - 2 * p2.x - p1.x) * (3 * cp2.x - 2 * p2.x - p1.x); 
    uint32_t vy = (3 * cp2.y - 2 * p2.y - p1.y) * (3 * cp2.y - 2 * p2.y - p1.y); 
    if (ux < vx) ux = vx; 
    if (uy < vy) uy = vy; 
    return ux + uy; 
} 
 
static int32_t QFlatness(const Vector2 p[3]) { 
    return ((int32_t) (p[1].x + 0.25*(p[0].x-p[1].x)+0.25*(p[2].x-p[1].x)) - (((int32_t) p[0].x + (int32_t) p[2].x) / 2) + 
            (int32_t) (p[1].y + 0.25*(p[0].y-p[1].y)+0.25*(p[2].y-p[1].y)) - (((int32_t) p[0].y + (int32_t) p[2].y) / 2)); 
}

Now, we can draw the curves like this: 

void CBezierCurve(const Vector2 p[4], const color_t color) { 
    if (CFlatness(p) < 2) { 
        _line(p[0], p[3], color); 
        return; 
    } else { 
        Vector2 split[2][4] = {{[3] = {0, 0}}, 
                               {[3] = {0, 0}}}; 
        CSplitCurve(p, split); 
        CBezierCurve(split[0], color); // recrusive 
        CBezierCurve(split[1], color); 
    } 
} 
 
void QBezierCurve(const Vector2 p[3], const color_t color) { 
    if (QFlatness(p) < 2) { 
        line(p[0], p[2], color); 
        return; 
    } else { 
        Vector2 split[2][3]; 
        QSplitCurve(p, split); 
        QBezierCurve(split[0], color); 
        QBezierCurve(split[1], color); 
    }
